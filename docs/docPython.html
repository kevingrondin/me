with tempfile.TemporaryFile('w+') as temp:
  temp.write("I'm a temp file!")
  # retourne au debut du fichier
  temp.seek(0)

Compresser

import os
import shutil
import zipfile

# ziper
with zipfile.ZipFile('first.zip', 'w') as z:
  z.write('file1.txt')

# lister les fichiers zipé
with zipfile.ZipFile('first.zip', 'r') as z:
  print(z.filelist)

# extraire 
with zipfile.ZipFile('first.zip', 'r') as z:
  # un fichier en particulier
  z.extract('file1.txt', 'extracted/file1.txt')

# compresser tout les fichiers
with zipfile.ZipFile('first.zip', 'w') as z:
  for f in os.listdir():
    z.write(f)

# extraire tout les fichiers
with zipfile.ZipFile('first.zip', 'r') as z:
  z.extractall('extracted/')

shutil.make_archive('mesfichiers.zip', 'zip')
shutil.unpack_archive('mesfichiers.zip.zip', extract_dir="folder")
shutil.make_archive('mesfichiers.zip.zip', 'tar')

Decorateur

class MyClass:
  def method(self):
    return 'instance method called', self
  
  @classmethod
  def classmethod(cls):
    return 'class method called', cls
  
  @staticmethod
  def staticmethod():
    return 'static method called'

obj = MyClass()
obj.method()
obj.classmethod()
obj.staticmethod()

MyClass.method() // missing argment
MyClass.classmethod()
MyClass.staticmethod()

@classmethod permet de définir des constructeur alternatif pour les classes

@staticmethod ne modifiera pas l'etat de la classe ou de l'instance, idéale pour des test, car elle est indépendante de la classe

class Pizza:
  def __init__(self, ingredients):
    self.ingredients = ingredients
  
  // this function is called when a method of class it's called 
  def __repr__(self):
    return f'Pizza({self.ingredients})'

  @classmethod
  def margherita(cls)
    return cls(['cheese', 'tomatoes'])

  def margherita(cls)

  • 1.1 Copie Simple: Copie tout les fichiers déposé sur le .bat

  • 1.2 Copie Avec Connexion: Copie tout les fichiers deposés sur le .bat, à l'aide d'un MOTDEPASSE et NOMUTILISATEUR
@echo off

set _SERVEUR        = \\SERVEUR
set _MOTDEPASS      = MOTDEPASSE
set _NOMUTILISATEUR = NOMUTILISATEUR
set _DESTINATION    = "\\DESTINATION\RESEAUX\DOSSIER\"

net use %_SERVEUR% /delete /yes

net use %_SERVEUR% %_MOTDEPASSE% /USER:%_NOMUTILISATEUR%
for %%F in (%*) do copy %%F %_DESTINATION%

  • 1.3 Deplace Fichier: Deplace les fichiers deposés sur le .bat, à l'aide d'un MOTDEPASSE et NOMUTILISATEUR
@echo off

set _SERVEUR         = \\SERVEUR
set _MOTDEPASSE      = MOTDEPASSE
set _NOMUTILISATEUR  = NOMUTILISATEUR
set _SOURCE          = \\SOURCE\fichier*
set _DESTINATION     = C:\DESTINATION\fichier*

net use %_SERVEUR% %_MOTDEPASSE% /USER:%_NOMUTILISATEUR%
xcopy %_SOURCE% %_DESTINATION%
del %_SOURCE%

Scanne

  • 2.1 Scanne Dir: scanne tout les fichiers d'un ou plusieurs repertoire
@echo off

SET _MP = MOTDEPASSE
SET _USER = \LOGIN
SET _DEST = D:\Mon\REPERTOIRE\

for %%M in (Rep01 Rep08 Rep10) do (
   net use \\SRV_%%M\ %_MP% /USER:%_USER%
   for %%F in ("\\SRV_%%M\Repetoire\source\*") do (
      echo %%F >> %DEST%[ScanneDir]DU%%M.txt
   )
)

TRIGGER

  • 4.1 Trigger:en tache planifier va scruter un dossier tout les X temmps
@echo off

setlocal enableextensions
setlocal enabledelayedexpansion

REM seul les variable TEMPORISATION TRIGGER DOSSIER LISTE MAIL peuvent t'être modifiée

REM Temportisation c'est le temps entre les deux fois ou il inspecte le dossier
set TEMPORISATION=6
REM Le modéle de fichier qu'on doit surveiller avec son chemin complet
set TRIGGER="\\SERVEUR\DOSSIER\fic*"
REM Le dossier du controle
set DOSSIER="C:\DOSSIER"
REM Le resultat des fichiers trouvé dans le dossier
set LISTE="C:\DOSSIER\liste.txt"
REM MAIL
set MAIL=destinataire@email.com -f emetteur@email.com

REM Auncune modification doit être aporté aprés cette ligne

dir /B %TRIGGER% > %LISTE%
cd %DOSSIER%
FOR /R %DOSSIER% %%i in (liste.txt) do (
    rem On teste si le fichier est vide
    if %%~zi GTR 0 (
        TIMEOUT %TEMPORISATION%
        dir /B %TRIGGER% > %LISTE%
        FOR /R %DOSSIER% %%i in (liste.txt) do (
            if %%~zi GTR 0 (
                C:\DOSSIER\prog\blat\blat.exe body.txt -to %MAIL% -s "OBJET CONTROLE" -server IPSERVEUR
            )
        )
    )
)

REM seul les variable RESULTAT DIR ENTETE FILESQL CONSTANTE DIRCONST peuvent t'être modifiée

REM Liste de constante à parcourire
set CONSTANTE="C:\API\liste_magasins.txt"


REM Auncune modification doit être aporté aprés cette ligne

cd %DIR%

FOR /F "tokens=1" %%j in(%CONSTANTE%) do (
    if "%%j"=="" GOTO EXIT

    sqlplus -s %%j/mp@hote @%FILESQL% %DIR%res_%%j.txt

    FOR /R "%DIR%" %%i in (res_%%j.txt) do (
        if %%~zi GTR 0 (

        )
    )
)

PROJET SEND-CSV-BYMAIL

  • 3.1 Mini projet
@echo off

setlocal enableextensions
setlocal enabledelayedexpansion

REM seul les variable RESULTAT DIR ENTETE FILESQL CONSTANTE DIRCONST peuvent t'être modifiée
REM mon entête du csv
set ENTET=C:\DOSSIER\_entete.txt
REM le corp du mail
set BOD=C:\DOSSIER\_body.txt
REM mon fichier resultat
set REP=C:\DOSSIER
REM mot de passe BDD
set MP=motdepasse
REM hote BDD
set HOTE=hote
REM Destinataire
set DEST=destinataire@mail.com

cd %REP%

rem on recupère tout nos elements pour boucler notre projet
FOR %%j IN (ENTITA ENTITB) DO (
    IF "%%j"=="" GOTO EXIT
    sqlplus -s %%j/%MP%@%HOTE% @%REP%\res_%%j.txt

    FOR /R "%REP%" %%i IN (res_%%j.txt) DO (
        IF %%~zi GTR 0 (
            TYPE %ENTET% > %REP%\res_%%j.csv
            TYPE %REP%\res_%%j.txt >> %REP%\res_%%j.csv
            rem envoyer un mail à chaque entitee
            if "%%j"=="ENTITA" %REP%\blat.exe %BOD% -to %DEST% -f crtl@control.com -s "Controle ENTITA" -attach %REP%\res_%%j.csv -server 192.168.0.172
        )
    )
)
